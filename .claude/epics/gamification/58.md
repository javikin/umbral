### 1. Companion Capture Logic

```kotlin
// CaptureCompanionUseCase.kt
class CaptureCompanionUseCase @Inject constructor(
    private val repository: ExpeditionRepository
) {
    suspend operator fun invoke(type: CompanionType): CaptureResult {
        // Check if already captured
        if (repository.hasCompanion(type)) {
            return CaptureResult.AlreadyCaptured
        }

        // Check requirements
        val progress = repository.getProgressSnapshot()
        val discoveredCount = repository.getDiscoveryCount()
        val companions = repository.getAllCompanionsSnapshot()

        val requirementMet = when (val req = type.captureRequirement) {
            is CaptureRequirement.Starter -> true
            is CaptureRequirement.Energy -> progress.totalEnergy >= req.amount
            is CaptureRequirement.Streak -> progress.longestStreak >= req.days
            is CaptureRequirement.Location -> repository.hasDiscoveredLocation(req.locationId)
            is CaptureRequirement.Locations -> discoveredCount >= req.count
            is CaptureRequirement.NightSessions -> progress.nightSessions >= req.count
            is CaptureRequirement.TotalDays -> progress.totalActiveDays >= req.days
        }

        if (!requirementMet) {
            return CaptureResult.RequirementNotMet(type.captureRequirement)
        }

        // Capture the companion
        val companion = CompanionEntity(
            id = UUID.randomUUID().toString(),
            type = type.name,
            name = null,
            evolutionState = 1,
            energyInvested = 0,
            capturedAt = System.currentTimeMillis(),
            isActive = companions.isEmpty() // First companion is auto-active
        )

        repository.insertCompanion(companion)
        return CaptureResult.Success(type)
    }
}

sealed class CaptureResult {
    data class Success(val type: CompanionType) : CaptureResult()
    object AlreadyCaptured : CaptureResult()
    data class RequirementNotMet(val requirement: CaptureRequirement) : CaptureResult()
}
```

### 2. Evolution System

```kotlin
// EvolveCompanionUseCase.kt
class EvolveCompanionUseCase @Inject constructor(
    private val repository: ExpeditionRepository
) {
    suspend operator fun invoke(companionId: String): EvolutionResult {
        val companion = repository.getCompanionById(companionId)
            ?: return EvolutionResult.NotFound

        // Check max evolution
        if (companion.evolutionState >= 3) {
            return EvolutionResult.MaxEvolution
        }

        // Calculate cost
        val cost = ExpeditionFormulas.evolutionCost(companion.evolutionState)

        // Check if enough energy invested
        if (companion.energyInvested < cost) {
            return EvolutionResult.InsufficientEnergy(
                required = cost,
                current = companion.energyInvested
            )
        }

        // Evolve
        repository.evolveCompanion(
            id = companionId,
            newState = companion.evolutionState + 1,
            remainingEnergy = companion.energyInvested - cost
        )

        return EvolutionResult.Success(
            companionType = CompanionType.valueOf(companion.type),
            newState = companion.evolutionState + 1
        )
    }
}

sealed class EvolutionResult {
    data class Success(val companionType: CompanionType, val newState: Int) : EvolutionResult()
    object NotFound : EvolutionResult()
    object MaxEvolution : EvolutionResult()
    data class InsufficientEnergy(val required: Int, val current: Int) : EvolutionResult()
}
```

```kotlin
// InvestEnergyUseCase.kt
class InvestEnergyUseCase @Inject constructor(
    private val repository: ExpeditionRepository
) {
    suspend operator fun invoke(companionId: String, amount: Int): InvestResult {
        val progress = repository.getProgressSnapshot()

        if (progress.totalEnergy < amount) {
            return InvestResult.InsufficientEnergy
        }

        val companion = repository.getCompanionById(companionId)
            ?: return InvestResult.CompanionNotFound

        if (companion.evolutionState >= 3) {
            return InvestResult.MaxEvolution
        }

        repository.investEnergy(companionId, amount)
        repository.spendEnergy(amount)

        val newTotal = companion.energyInvested + amount
        val cost = ExpeditionFormulas.evolutionCost(companion.evolutionState)

        return InvestResult.Success(
            newTotal = newTotal,
            evolutionReady = newTotal >= cost
        )
    }
}
```

```kotlin
@Composable
fun CompanionListScreen(
    viewModel: CompanionViewModel = hiltViewModel(),
    onCompanionClick: (String) -> Unit,
    onBack: () -> Unit
) {
    val uiState by viewModel.uiState.collectAsState()

    Scaffold(
        topBar = {
            TopAppBar(
                title = { Text("Compañeros") },
                navigationIcon = {
                    IconButton(onClick = onBack) {
                        Icon(Icons.Default.ArrowBack, "Volver")
                    }
                }
            )
        }
    ) { padding ->
        LazyVerticalGrid(
            columns = GridCells.Fixed(2),
            modifier = Modifier.padding(padding),
            contentPadding = PaddingValues(16.dp),
            horizontalArrangement = Arrangement.spacedBy(12.dp),
            verticalArrangement = Arrangement.spacedBy(12.dp)
        ) {
            items(CompanionType.values()) { type ->
                val captured = uiState.companions.find { it.type == type.name }
                val canCapture = uiState.availableCaptures.contains(type)

                CompanionGridCard(
                    type = type,
                    captured = captured,
                    canCapture = canCapture,
                    onClick = {
                        if (captured != null) {
                            onCompanionClick(captured.id)
                        } else if (canCapture) {
                            viewModel.captureCompanion(type)
                        }
                    }
                )
            }
        }
    }
}

@Composable
fun CompanionGridCard(
    type: CompanionType,
    captured: CompanionEntity?,
    canCapture: Boolean,
    onClick: () -> Unit
) {
    val isCaptured = captured != null
    val isLocked = !isCaptured && !canCapture

    UmbralCard(
        onClick = if (!isLocked) onClick else null,
        modifier = Modifier
            .fillMaxWidth()
            .aspectRatio(0.8f),
        elevation = if (isCaptured) UmbralElevation.Medium else UmbralElevation.Low
    ) {
        Column(
            modifier = Modifier
                .fillMaxSize()
                .padding(12.dp),
            horizontalAlignment = Alignment.CenterHorizontally,
            verticalArrangement = Arrangement.Center
        ) {
            // Companion image placeholder
            Box(
                modifier = Modifier
                    .size(80.dp)
                    .clip(CircleShape)
                    .background(
                        when {
                            isCaptured -> MaterialTheme.colorScheme.primaryContainer
                            canCapture -> MaterialTheme.colorScheme.secondaryContainer
                            else -> MaterialTheme.colorScheme.surfaceVariant
                        }
                    ),
                contentAlignment = Alignment.Center
            ) {
                if (isLocked) {
                    Icon(
                        imageVector = Icons.Default.Lock,
                        contentDescription = null,
                        tint = MaterialTheme.colorScheme.outline
                    )
                } else {
                    // TODO: Replace with actual companion image
                    Text(
                        text = type.displayName.first().toString(),
                        style = MaterialTheme.typography.headlineMedium,
                        fontWeight = FontWeight.Bold
                    )
                }
            }

            Spacer(modifier = Modifier.height(8.dp))

            Text(
                text = if (isCaptured || canCapture) type.displayName else "???",
                style = MaterialTheme.typography.titleSmall,
                fontWeight = FontWeight.Bold
            )

            if (isCaptured) {
                Text(
                    text = "Nivel ${captured!!.evolutionState}",
                    style = MaterialTheme.typography.labelSmall,
                    color = MaterialTheme.colorScheme.primary
                )

                // Evolution progress
                if (captured.evolutionState < 3) {
                    val progress = captured.energyInvested.toFloat() /
                        ExpeditionFormulas.evolutionCost(captured.evolutionState)
                    LinearProgressIndicator(
                        progress = { progress.coerceIn(0f, 1f) },
                        modifier = Modifier
                            .fillMaxWidth()
                            .padding(top = 4.dp)
                    )
                }
            } else if (canCapture) {
                Text(
                    text = "¡Disponible!",
                    style = MaterialTheme.typography.labelSmall,
                    color = MaterialTheme.colorScheme.secondary
                )
            } else {
                Text(
                    text = getRequirementHint(type.captureRequirement),
                    style = MaterialTheme.typography.labelSmall,
                    color = MaterialTheme.colorScheme.outline,
                    textAlign = TextAlign.Center,
                    maxLines = 2
                )
            }
        }
    }
}

private fun getRequirementHint(requirement: CaptureRequirement): String = when (requirement) {
    is CaptureRequirement.Starter -> ""
    is CaptureRequirement.Energy -> "${requirement.amount} energía total"
    is CaptureRequirement.Streak -> "${requirement.days} días de racha"
    is CaptureRequirement.Location -> "Descubre locación especial"
    is CaptureRequirement.Locations -> "${requirement.count} locaciones"
    is CaptureRequirement.NightSessions -> "${requirement.count} sesiones nocturnas"
    is CaptureRequirement.TotalDays -> "${requirement.days} días de uso"
}
```

```kotlin
// PassiveBonusCalculator.kt
class PassiveBonusCalculator @Inject constructor(
    private val repository: ExpeditionRepository
) {
    suspend fun getEnergyMultiplier(): Float {
        val activeCompanion = repository.getActiveCompanionSnapshot() ?: return 1f
        val type = CompanionType.valueOf(activeCompanion.type)

        return when (type) {
            CompanionType.TERRA -> 1.05f
            CompanionType.IGNIS -> if (repository.getCurrentStreak() >= 5) 1.10f else 1f
            CompanionType.NOCTIS -> if (isNightTime()) 2f else 1f
            else -> 1f
        }
    }

    suspend fun getXpMultiplier(): Float {
        val activeCompanion = repository.getActiveCompanionSnapshot() ?: return 1f
        val type = CompanionType.valueOf(activeCompanion.type)

        return when (type) {
            CompanionType.LUMINA -> 1.05f
            else -> 1f
        }
    }

    private fun isNightTime(): Boolean {
        val hour = Calendar.getInstance().get(Calendar.HOUR_OF_DAY)
        return hour >= 22 || hour < 6
    }
}
```
