### 2. CompanionAnimation Component

```kotlin
@Composable
fun CompanionAnimation(
    type: CompanionType,
    state: CompanionAnimationState,
    modifier: Modifier = Modifier
) {
    val composition by rememberLottieComposition(
        spec = LottieCompositionSpec.Asset(
            when (state) {
                CompanionAnimationState.IDLE -> LottieAnimations.companionIdle(type)
                CompanionAnimationState.HAPPY -> LottieAnimations.companionHappy(type)
                CompanionAnimationState.EVOLVING -> LottieAnimations.EVOLUTION
            }
        )
    )

    val progress by animateLottieCompositionAsState(
        composition = composition,
        iterations = when (state) {
            CompanionAnimationState.IDLE -> LottieConstants.IterateForever
            CompanionAnimationState.HAPPY -> 1
            CompanionAnimationState.EVOLVING -> 1
        },
        restartOnPlay = state != CompanionAnimationState.IDLE
    )

    LottieAnimation(
        composition = composition,
        progress = { progress },
        modifier = modifier
    )
}

enum class CompanionAnimationState {
    IDLE, HAPPY, EVOLVING
}
```

```kotlin
@Composable
fun CompanionDetailScreen(
    companionId: String,
    viewModel: CompanionDetailViewModel = hiltViewModel(),
    onBack: () -> Unit
) {
    val uiState by viewModel.uiState.collectAsState()
    var showEvolutionDialog by remember { mutableStateOf(false) }
    var animationState by remember { mutableStateOf(CompanionAnimationState.IDLE) }

    LaunchedEffect(companionId) {
        viewModel.loadCompanion(companionId)
    }

    val companion = uiState.companion ?: return
    val type = CompanionType.valueOf(companion.type)

    Scaffold(
        topBar = {
            TopAppBar(
                title = { Text(companion.name ?: type.displayName) },
                navigationIcon = {
                    IconButton(onClick = onBack) {
                        Icon(Icons.Default.ArrowBack, "Volver")
                    }
                },
                actions = {
                    if (companion.isActive) {
                        Icon(
                            imageVector = Icons.Default.Star,
                            contentDescription = "Activo",
                            tint = MaterialTheme.colorScheme.primary
                        )
                    }
                }
            )
        }
    ) { padding ->
        Column(
            modifier = Modifier
                .fillMaxSize()
                .padding(padding)
                .verticalScroll(rememberScrollState()),
            horizontalAlignment = Alignment.CenterHorizontally
        ) {
            // Companion animation
            Box(
                modifier = Modifier
                    .fillMaxWidth()
                    .height(250.dp)
                    .background(MaterialTheme.colorScheme.surfaceVariant),
                contentAlignment = Alignment.Center
            ) {
                CompanionAnimation(
                    type = type,
                    state = animationState,
                    modifier = Modifier.size(200.dp)
                )
            }

            Spacer(modifier = Modifier.height(24.dp))

            // Info section
            Column(
                modifier = Modifier.padding(horizontal = 24.dp),
                horizontalAlignment = Alignment.CenterHorizontally
            ) {
                // Element badge
                Surface(
                    color = MaterialTheme.colorScheme.secondaryContainer,
                    shape = RoundedCornerShape(16.dp)
                ) {
                    Text(
                        text = type.element,
                        modifier = Modifier.padding(horizontal = 16.dp, vertical = 8.dp),
                        style = MaterialTheme.typography.labelLarge
                    )
                }

                Spacer(modifier = Modifier.height(16.dp))

                // Evolution state
                Text(
                    text = "Estado de Evolución",
                    style = MaterialTheme.typography.titleMedium
                )
                EvolutionIndicator(
                    currentState = companion.evolutionState,
                    modifier = Modifier.padding(vertical = 8.dp)
                )

                // Energy invested
                if (companion.evolutionState < 3) {
                    val evolutionCost = ExpeditionFormulas.evolutionCost(companion.evolutionState)
                    val progress = companion.energyInvested.toFloat() / evolutionCost

                    Text(
                        text = "Energía: ${companion.energyInvested} / $evolutionCost",
                        style = MaterialTheme.typography.bodyMedium
                    )

                    LinearProgressIndicator(
                        progress = { progress },
                        modifier = Modifier
                            .fillMaxWidth()
                            .padding(vertical = 8.dp)
                    )

                    // Invest energy button
                    OutlinedButton(
                        onClick = { viewModel.investEnergy(100) },
                        enabled = uiState.currentEnergy >= 100
                    ) {
                        Icon(Icons.Default.Bolt, null)
                        Spacer(modifier = Modifier.width(8.dp))
                        Text("Invertir 100 Energía")
                    }

                    Spacer(modifier = Modifier.height(8.dp))

                    // Evolve button
                    if (companion.energyInvested >= evolutionCost) {
                        Button(
                            onClick = { showEvolutionDialog = true },
                            colors = ButtonDefaults.buttonColors(
                                containerColor = MaterialTheme.colorScheme.tertiary
                            )
                        ) {
                            Icon(Icons.Default.AutoAwesome, null)
                            Spacer(modifier = Modifier.width(8.dp))
                            Text("¡Evolucionar!")
                        }
                    }
                } else {
                    Text(
                        text = "Evolución máxima alcanzada",
                        style = MaterialTheme.typography.bodyMedium,
                        color = MaterialTheme.colorScheme.primary
                    )
                }

                Spacer(modifier = Modifier.height(24.dp))

                // Passive ability
                UmbralCard(
                    modifier = Modifier.fillMaxWidth()
                ) {
                    Column(modifier = Modifier.padding(16.dp)) {
                        Text(
                            text = "Habilidad Pasiva",
                            style = MaterialTheme.typography.titleSmall,
                            fontWeight = FontWeight.Bold
                        )
                        Spacer(modifier = Modifier.height(4.dp))
                        Text(
                            text = type.passiveBonus,
                            style = MaterialTheme.typography.bodyMedium,
                            color = MaterialTheme.colorScheme.primary
                        )
                    }
                }

                Spacer(modifier = Modifier.height(16.dp))

                // Set as active button
                if (!companion.isActive) {
                    Button(
                        onClick = { viewModel.setAsActive() },
                        modifier = Modifier.fillMaxWidth()
                    ) {
                        Text("Establecer como activo")
                    }
                }
            }
        }
    }

    // Evolution dialog
    if (showEvolutionDialog) {
        EvolutionDialog(
            companionType = type,
            currentState = companion.evolutionState,
            onConfirm = {
                animationState = CompanionAnimationState.EVOLVING
                viewModel.evolve()
                showEvolutionDialog = false
            },
            onDismiss = { showEvolutionDialog = false }
        )
    }
}

@Composable
fun EvolutionIndicator(
    currentState: Int,
    modifier: Modifier = Modifier
) {
    Row(
        modifier = modifier,
        horizontalArrangement = Arrangement.spacedBy(8.dp),
        verticalAlignment = Alignment.CenterVertically
    ) {
        repeat(3) { index ->
            val state = index + 1
            val isActive = state <= currentState

            Box(
                modifier = Modifier
                    .size(40.dp)
                    .clip(CircleShape)
                    .background(
                        if (isActive) MaterialTheme.colorScheme.primary
                        else MaterialTheme.colorScheme.surfaceVariant
                    ),
                contentAlignment = Alignment.Center
            ) {
                Text(
                    text = "$state",
                    color = if (isActive) MaterialTheme.colorScheme.onPrimary
                            else MaterialTheme.colorScheme.onSurfaceVariant,
                    fontWeight = FontWeight.Bold
                )
            }

            if (index < 2) {
                Divider(
                    modifier = Modifier.width(24.dp),
                    color = if (state < currentState) MaterialTheme.colorScheme.primary
                            else MaterialTheme.colorScheme.surfaceVariant
                )
            }
        }
    }
}
```

### 4. EvolutionDialog

```kotlin
@Composable
fun EvolutionDialog(
    companionType: CompanionType,
    currentState: Int,
    onConfirm: () -> Unit,
    onDismiss: () -> Unit
) {
    var isAnimating by remember { mutableStateOf(false) }

    val composition by rememberLottieComposition(
        spec = LottieCompositionSpec.Asset(LottieAnimations.EVOLUTION)
    )

    Dialog(onDismissRequest = { if (!isAnimating) onDismiss() }) {
        Card(
            modifier = Modifier
                .fillMaxWidth()
                .padding(16.dp),
            shape = RoundedCornerShape(24.dp)
        ) {
            Column(
                modifier = Modifier.padding(24.dp),
                horizontalAlignment = Alignment.CenterHorizontally
            ) {
                if (isAnimating) {
                    // Evolution animation
                    LottieAnimation(
                        composition = composition,
                        iterations = 1,
                        modifier = Modifier.size(200.dp)
                    )

                    LaunchedEffect(Unit) {
                        delay(3000) // Animation duration
                        onConfirm()
                    }
                } else {
                    Icon(
                        imageVector = Icons.Default.AutoAwesome,
                        contentDescription = null,
                        modifier = Modifier.size(64.dp),
                        tint = MaterialTheme.colorScheme.primary
                    )

                    Spacer(modifier = Modifier.height(16.dp))

                    Text(
                        text = "¿Evolucionar a ${companionType.displayName}?",
                        style = MaterialTheme.typography.titleLarge,
                        textAlign = TextAlign.Center
                    )

                    Spacer(modifier = Modifier.height(8.dp))

                    Text(
                        text = "Estado $currentState → Estado ${currentState + 1}",
                        style = MaterialTheme.typography.bodyMedium,
                        color = MaterialTheme.colorScheme.onSurfaceVariant
                    )

                    Spacer(modifier = Modifier.height(24.dp))

                    Row(
                        modifier = Modifier.fillMaxWidth(),
                        horizontalArrangement = Arrangement.spacedBy(12.dp)
                    ) {
                        OutlinedButton(
                            onClick = onDismiss,
                            modifier = Modifier.weight(1f)
                        ) {
                            Text("Cancelar")
                        }

                        Button(
                            onClick = { isAnimating = true },
                            modifier = Modifier.weight(1f)
                        ) {
                            Text("¡Evolucionar!")
                        }
                    }
                }
            }
        }
    }
}
```
